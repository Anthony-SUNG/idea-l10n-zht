報告 Spring XML 應用程序上下文中的停用配置文件。

示例：


  <!-- 為此應用程序上下文選擇有效配置文件 'DEV' -->
  <beans profile="DEV">
    <bean class="MyFactory" factory-method="createBean"/>
  </beans>
  <!--  此標記的內容高亮顯示為 'unused' -->
  <beans profile="TESTS">
    <bean class="MyTestFactory" factory-method="createBean"/>
  </beans>
  <!--  'unused' 部分的結尾 -->
  </beans>
